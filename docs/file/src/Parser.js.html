    import stream from 'stream'
import Utilities from './Utilities.js'

const Transform = stream.Transform

class Parser extends Transform {

  constructor () {

    super({
      'objectMode': true
    })

    this.lineNumber = 1
    this.numberOfPaths = null
    this.numberOfPatterns = null
  }

  nextLine () {

    this.lineNumber++
  }

  _transform (chunk, encoding, processed) {

    if (format.isNumberOfPatternsLine(this.lineNumber)) {

      this.numberOfPatterns = Utilities.parseNumber(chunk)
    }

    else if (format.isPatternLine(this.lineNumber, this.numberOfPatterns)) {

      this.push({patterns: {[chunk]: chunk}})
    }

    else if (format.isNumberOfPathsLine(this.lineNumber, this.numberOfPatterns)) {

      this.numberOfPaths = Utilities.parseNumber(chunk)
    }

    else if (format.isPathLine(this.lineNumber, this.numberOfPatterns, this.numberOfPaths)) {

      this.push({paths: {[chunk]: chunk}})
    }

    this.nextLine()

    processed()
  }
}

export default Parser
